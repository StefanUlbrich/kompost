20use std::marker::PhantomData;
21
22/// Iterator that with a context and a closures called in the [`ComposedIterator::new`] and
23/// [`ComposedIterator::next`] methods. Used in the [`ComposedIterable`] trait.
24///
25/// [`crate::anonymous::AnonymousIterator`] could not be used as a constant closure cannot be used within
26/// the implementation of [`ComposedIterable`].
27pub struct ComposedIterator<IterIn, IterOut, In, Out, Init>
28where
29    Init: FnOnce(IterIn) -> IterOut,
30
31    IterIn: Iterator<Item = In>,
32    IterOut: Iterator<Item = Out>,
33{
34    iter: IterOut,
35    _iter: PhantomData<IterIn>,
36    _init: PhantomData<Init>,
37}
38
39impl<IterIn, IterOut, In, Out, Init> ComposedIterator<IterIn, IterOut, In, Out, Init>
40where
41    Init: FnOnce(IterIn) -> IterOut,
42
43    IterIn: Iterator<Item = In>,
44    IterOut: Iterator<Item = Out>,
45{
46    pub fn new(iter: IterIn, init_fn: Init) -> Self {
47        Self {
48            iter: init_fn(iter),
49            _iter: PhantomData,
50            _init: PhantomData,
51        }
52    }
53}
54
55impl<IterIn, IterOut, In, Out, Init> Iterator for ComposedIterator<IterIn, IterOut, In, Out, Init>
56where
57    Init: FnOnce(IterIn) -> IterOut,
58
59    IterIn: Iterator<Item = In>,
60    IterOut: Iterator<Item = Out>,
61{
62    type Item = Out;
63
64    fn next(&mut self) -> Option<Self::Item> {
65        self.iter.next()
66    }
67}
68pub trait ComposedIterable<IterIn, IterOut, In, Out, Init>
69where
70    Init: FnOnce(IterIn) -> IterOut,
71
72    IterIn: Iterator<Item = In>,
73    IterOut: Iterator<Item = Out>,
74{
75    /// Allows defining grouping frequently used [`Iterator`] methods (such as `map` or `scan`)
76    /// and reuse them.
77    ///
78    /// ## Arguments
79    ///
80    /// * `init_fn`: The closure that receives the current [`IntoIterator`] and produces the initial
81    ///   context.
82    ///
83    /// ## Examples
84    ///
85    /// ```
86    /// use kompost::ComposedIterable;
87    ///
88    /// fn favorite_pipeline(it: impl Iterator<Item = i32>) -> impl Iterator<Item = f64> {
89    ///     it.skip(5)
90    ///         .map(|x| x.pow(2))
91    ///         .take_while(|x| *x < 100)
92    ///         .map(|x| x as f64)
93    /// }
94    ///
95    /// assert_eq!(
96    ///     [1, 2, 3, 4, 5, 6, 7].into_iter().composed(favorite_pipeline).collect::<Vec<_>>(),
97    ///     vec![36.0f64, 49.0]
98    /// )
99    /// ```
100    fn composed(self, init_fs: Init) -> ComposedIterator<IterIn, IterOut, In, Out, Init>;
101}
102
103impl<IterIn, IterOut, In, Out, Init> ComposedIterable<IterIn, IterOut, In, Out, Init> for IterIn
104where
105    Init: FnOnce(IterIn) -> IterOut,
106
107    IterIn: Iterator<Item = In>,
108    IterOut: Iterator<Item = Out>,
109{
110    fn composed(self, init_fn: Init) -> ComposedIterator<IterIn, IterOut, In, Out, Init> {
111        ComposedIterator::new(self, init_fn)
112    }
