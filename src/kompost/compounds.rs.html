22/// Function to be used with the [`crate::Composed::composed`] method.
23/// It transposes an [`IntoIterator`] of [`std::slice`], a data structure often encountered
24/// when storing 2D arrays in a single (row-major) array.
25///
26/// ## Example
27///
28/// ```
29/// # use kompost::{Composed, compounds::transpose};
30/// let x: Vec<_> = [1, 2, 3, 4]                 // An array in row-major order
31///     .chunks(2)                               // Only defined on slices and vectors
32///     .composed(transpose)
33///     .collect();
34/// assert_eq!(x, [1, 3, 2, 4]);
35/// ```
36pub fn transpose<'a, T: 'a + Copy>(iter: impl Iterator<Item = &'a [T]>) -> impl Iterator<Item = T> {
37    iter.into_iter()
38        .anonymous(
39            |chunks| chunks.map(|row| row.iter()).collect::<Vec<_>>(),
40            |context| {
41                let transposed = context // &mut Vec<Iter,i32>
42                    .iter_mut()
43                    .filter_map(Iterator::next) // impl Iterator<Item = &i32>
44                    // .filter_map(|i| i.next()) // impl Iterator<Item = &i32>
45                    .collect::<Vec<_>>(); // Vec<&i32>
46                                          // If the iterators over the rows return `None`, transpose is empty
47                if transposed.is_empty() {
48                    None
49                } else {
50                    Some(transposed.into_iter())
51                }
52            },
53        ) // AnonymousIterator
54        .flatten() // impl Iterator<Item = &i32>
55        .copied() // impl Iterator<Item = i32>
56}

58/// A compound function to be used with the [`crate::Composed::composed`] method that takes
59/// an additional single `usize` as a parameter and computes a window of that size for *every element*
60/// of the iterator (periodic).
61///
62/// This is requires to write an additional closure when it is used, but this might change in the future
63/// when a functor trait might be written instead.
64///
65/// ## Example
66///
67/// ```
68/// # use kompost::{Composed, compounds::periodic_windows};
69/// let size=3;
70/// let x = [1, 2, 3, 4].into_iter()
71///     .composed(|i| periodic_windows(3, i))
72///     .flatten()
73///     .collect::<Vec<_>>();
74/// assert_eq!(x, [1,2,3,2,3,4,3,4,1,4,1,2])
75/// ```
76pub fn periodic_windows<T>(
77    size: usize,
78    it: impl ExactSizeIterator<Item = T> + Clone,
79) -> impl Iterator<Item = impl Iterator<Item = T>> {
80    it.anonymous(
81        |it| {
82            let len = it.len();
83            (0usize, len, it.cycle())
84            // it.take(len)
85        },
86        move |(i, len, it)| {
87            let window = it.clone();
88            it.next();
89            *i += 1;
90            if i <= len {
91                Some(window.take(size))
92            } else {
93                None
94            }
95        },
96    )
97}
