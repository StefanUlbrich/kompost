26/// Function to be used with the [`crate::Composed::composed`] method.
27/// It transposes an [`IntoIterator`] of [`std::slice`], a data structure often encountered
28/// when storing 2D arrays in a single (row-major) array and using
29/// [`chunks`](slice::chunks) for iteration
30///
31/// ## Example
32///
33/// ```
34/// # use kompost::{Composed, compounds::transpose_slice};
35/// let x: Vec<_> = [1, 2, 3, 4]                 // An array in row-major order
36///     .chunks(2)                               // Only defined on slices and vectors
37///     .composed(transpose_slice)
38///     .collect();
39/// assert_eq!(x, [1, 3, 2, 4]);
40/// ```
41pub fn transpose_slice<'a, T: 'a + Copy>(
42    iter: impl Iterator<Item = &'a [T]>,
43) -> impl Iterator<Item = T> {
44    iter.into_iter()
45        .anonymous(
46            |chunks| chunks.map(|row| row.iter()).collect::<Vec<_>>(),
47            |context| {
48                let transposed = context // &mut Vec<Iter,i32>
49                    .iter_mut()
50                    .filter_map(Iterator::next) // impl Iterator<Item = &i32>
51                    // .filter_map(|i| i.next()) // impl Iterator<Item = &i32>
52                    .collect::<Vec<_>>(); // Vec<&i32>
53                                          // If the iterators over the rows return `None`, transpose is empty
54                if transposed.is_empty() {
55                    None
56                } else {
57                    Some(transposed.into_iter())
58                }
59            },
60        ) // AnonymousIterator
61        .flatten() // impl Iterator<Item = &i32>
62        .copied() // impl Iterator<Item = i32>
63}

66/// TBD: Transpose over Iterable of Iterable
67///
68/// ```rust
69/// use kompost::*;
70/// use kompost::compounds::*;
71///
72/// let a = [1, 2, 3];
73/// let b = [4, 5, 6];
74/// let c = [a.iter(), b.iter()];
75/// let d = c
76///     .into_iter()
77///     .composed(transpose)
78///     .flatten()
79///     .copied()
80///     .collect::<Vec<_>>();
81/// assert_eq!(d, [1,4,2,5,3,6]);
82/// ```
83pub fn transpose<T>(
84    iter: impl Iterator<Item = impl Iterator<Item = T>>,
85) -> impl Iterator<Item = impl Iterator<Item = T>> {
86    iter.into_iter().anonymous(
87        |rows| rows.collect::<Vec<_>>(),
88        |context| {
89            let transposed = context
90                .iter_mut()
91                .filter_map(Iterator::next)
92                .collect::<Vec<_>>();
93            if transposed.is_empty() {
94                None
95            } else {
96                Some(transposed.into_iter())
97            }
98        },
99    )
100}

102/// A compound function to be used with the [`crate::Composed::composed`] method that takes
103/// an additional single `usize` as a parameter and computes a window of that size for *every element*
104/// of the iterator (circular, it takes elements from the beginning for later windows).
105///
106/// This is requires to write an additional closure when it is used, but this might change in the future
107/// when a functor trait might be written instead.
108///
109/// ## Example
110///
111/// ```
112/// # use kompost::{Composed, compounds::circular_windows};
113/// let size=3;
114/// let x = [1, 2, 3, 4].into_iter()
115///     .composed(|i| circular_windows(3, i))
116///     .flatten()
117///     .collect::<Vec<_>>();
118/// assert_eq!(x, [1,2,3,2,3,4,3,4,1,4,1,2])
119/// ```
120pub fn circular_windows<T>(
121    size: usize,
122    it: impl ExactSizeIterator<Item = T> + Clone,
123) -> impl Iterator<Item = impl Iterator<Item = T>> {
124    it.anonymous(
125        |it| {
126            let len = it.len();
127            (0usize, len, it.cycle())
128            // it.take(len)
129        },
130        move |(i, len, it)| {
131            let window = it.clone();
132            it.next();
133            *i += 1;
134            if i <= len {
135                Some(window.take(size))
136            } else {
137                None
138            }
139        },
140    )
141}

144/// Compound function to generate circular sliding windows over a 2D data structure
145/// in form of an [`Iterator`] over slices (such as returned by the [`chunks`](slice::chunks) method)
146/// See [this example](crate#complex-example section) for how to use it.
147pub fn circular_windows_2d_slice<'a, T: 'a>(
148    it: impl ExactSizeIterator<Item = &'a [T]> + Clone,
149    size_m: usize,
150    size_n: usize,
151) -> impl Iterator<Item = impl Iterator<Item = impl Iterator<Item = impl Iterator<Item = &'a T>>>> {
152    it.composed(move |it| circular_windows(size_m, it))
153        .map(move |rows| {
154            rows.map(move |row| {
155                row.into_iter()
156                    .composed(move |it| circular_windows(size_n.clone(), it))
157            })
158            .composed(transpose)
159        })
160}

162/// Compound function to generate circular sliding windows over a 2D data structure
163/// in form of an [`Iterator`] over [`Iterator`].
164/// See [this example](crate#complex-example section) for how to use it.
165pub fn circular_windows_2d<T>(
166    it: impl ExactSizeIterator<Item = impl ExactSizeIterator<Item = T> + Clone> + Clone,
167    size_m: usize,
168    size_n: usize,
169) -> impl Iterator<Item = impl Iterator<Item = impl Iterator<Item = impl Iterator<Item = T>>>> {
170    it.composed(move |it| circular_windows(size_m, it))
171        .map(move |rows| {
172            rows.map(move |row| row.composed(move |it| circular_windows(size_n.clone(), it)))
173                .composed(transpose)
174        })
