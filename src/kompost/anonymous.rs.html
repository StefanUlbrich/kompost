22/// Iterator that with a context and a closures called in the [`AnonymousIterator::new`] and
23/// [`AnonymousIterator::next`] methods.
24pub struct AnonymousIterator<Iter, In, Out, Init, Next, Context>
25where
26    Init: FnOnce(Iter) -> Context,
27    Next: Fn(&mut Context) -> Option<Out>,
28    Iter: Iterator<Item = In>,
29{
30    // init_fn: Init,
31    next_fn: Next,
32    context: Context,
33    _iter: PhantomData<Iter>,
34    _init: PhantomData<Init>,
35}
36
37impl<Iter, In, Out, Init, Next, Context> AnonymousIterator<Iter, In, Out, Init, Next, Context>
38where
39    Init: FnOnce(Iter) -> Context,
40    Next: Fn(&mut Context) -> Option<Out>,
41    Iter: Iterator<Item = In>,
42{
43    pub fn new(iter: Iter, init_fn: Init, next_fn: Next) -> Self {
44        Self {
45            next_fn,
46            context: init_fn(iter),
47            _iter: PhantomData,
48            _init: PhantomData,
49        }
50    }
51}
52
53impl<Iter, In, Out, Init, Next, Context> Iterator
54    for AnonymousIterator<Iter, In, Out, Init, Next, Context>
55where
56    Init: FnOnce(Iter) -> Context,
57    Next: Fn(&mut Context) -> Option<Out>,
58    Iter: Iterator<Item = In>,
59{
60    type Item = Out;
61
62    fn next(&mut self) -> Option<Self::Item> {
63        (self.next_fn)(&mut self.context)
64    }
65}

67/// Trait that adds a method to [`Iterator`] structs (i.e., [`Iterator`]s, [`slice`](std::slice)s, and [`Vec`]s) that
68/// allows painlessly create anonynmous iterators for one-time use.
69pub trait Anonymous<Iter, In, Out, Init, Next, Context>
70where
71    Init: FnOnce(Iter) -> Context,
72    Next: Fn(&mut Context) -> Option<Out>,
73    Iter: Iterator<Item = In>,
74{
75    /// Creates an anonymous iterator.
76    ///
77    /// The iterator is defined by its behavior upon creation ([`AnonymousIterator::new`]) and when
78    /// [`AnonymousIterator::next`] are called. The former is set by a closure that receives the current
79    /// [`IntoIterator`] and returns a context (can consume data from the scope). The second closure
80    /// computes the next value from the context. The [`anonymous()`](Anonymous::anonymous)
81    /// method resembles [`Iterator::scan()`](Iterator::scan) with the exception that the context
82    /// may depend on the calling [`Iterator`].
83    ///
84    /// ## Arguments
85    ///
86    /// * `init_fn`: The closure that receives the current [`Iterator`] and produces the initial
87    ///   context.
88    /// * `next_fn`: The closure that produces the next value from a mutable reference to the context.
89    fn anonymous(
90        self,
91        init_fn: Init,
92        next_fn: Next,
93    ) -> AnonymousIterator<Iter, In, Out, Init, Next, Context>;
94}
95
96impl<Iter, In, Out, Init, Next, Context> Anonymous<Iter, In, Out, Init, Next, Context> for Iter
97where
98    Init: FnOnce(Iter) -> Context,
99    Next: Fn(&mut Context) -> Option<Out>,
100    Iter: Iterator<Item = In>,
101{
102    fn anonymous(
103        self,
104        init_fn: Init,
105        next_fn: Next,
106    ) -> AnonymousIterator<Iter, In, Out, Init, Next, Context> {
107        AnonymousIterator::new(self, init_fn, next_fn)
108    }
